## 项目概述

本项目是一个功能完整的五子棋在线对战平台，采用前后端分离架构：

- **前端（推荐）**: Vue 3 + TypeScript + Vite 现代化架构（frontend/）
- **前端（原始）**: 纯 HTML/CSS/JavaScript 单页应用（oldweb.disabled/，已弃用）
- **后端**: Node.js + WebSocket 服务器（server/server.js）
- **AI 引擎**: 基于 Alpha-Beta 剪枝算法（ai.js，已迁移到 TypeScript）
- **AI Worker**: Web Worker 包装器（ai-worker.js，已迁移到 TypeScript）

```typescript
  gameMode.value = mode;
```

```typescript
// frontend/src/App.vue

function startAIGame(config: {
  playerName: string;
  playerColor: Player;
  difficulty: Difficulty;
  size: number;
  time: number
}) {
  gameStore.myColor = config.playerColor;
  gameStore.myName = config.playerName;
  gameStore.aiColor = (config.playerColor === 1 ? 2 : 1) as Player;
  gameStore.undoLimit = 999; // AI 模式允许无限悔棋

  const aiName = config.difficulty === 'easy' ? 'AI(简单)' :
                  config.difficulty === 'medium' ? 'AI(中等)' : 'AI(困难)';

  gameStore.startGame('ai', {
    player1: config.playerColor === 1 ? config.playerName : aiName,
    player2: config.playerColor === 2 ? config.playerName : aiName,
    time: config.time,
    size: config.size
  });

  // 初始化 AI 引擎
  initAI(config.difficulty);
  gameStore.setOnAITurn(() => aiMakeMove());

  // 如果 AI 执黑（先手），让 AI 先下
  if (gameStore.aiColor === 1) {
    setTimeout(() => aiMakeMove(), 300);
  }

  currentScreen.value = 'game';
}
```

**AI 思考流程（useAI Composable）**:
```typescript
// frontend/src/composables/useAI.ts

export function useAI() {
  const isThinking = ref(false);
  const aiWorker = ref<Worker | null>(null);
  const requestCounter = ref(0);
  const currentRequestId = ref<number | null>(null);

  function initAI(difficulty: Difficulty) {
    // 创建 Web Worker
    aiWorker.value = new Worker(new URL('../workers/ai-worker.ts', import.meta.url), {
      type: 'module'
    });

    aiWorker.value.onmessage = (e) => {
      const result = e.data;

      // 检查请求 ID 是否匹配（忽略过期响应）
      if (result.requestId !== currentRequestId.value) {
        console.log('AI: 忽略过期响应');
        return;
      }

      if (result.success && result.move) {
        const thinkTime = Math.ceil(result.thinkTime / 1000);

        // 扣除 AI 的思考时间
        if (thinkTime > 0 && gameStore.aiColor !== null) {
          gameStore.players[gameStore.aiColor].time = Math.max(0,
            gameStore.players[gameStore.aiColor].time - thinkTime);

          // 检查 AI 是否超时
          if (gameStore.players[gameStore.aiColor].time <= 0) {
            const winner = gameStore.aiColor === 1 ? 2 : 1;
            gameStore.endGame(winner);
            return;
          }
        }

        // 执行 AI 落子
        gameStore.doPlacePiece(result.move.row, result.move.col, gameStore.aiColor!);
      }

      isThinking.value = false;
    };
  }

  function makeMove() {
    if (isThinking.value || !aiWorker.value) return;

    const requestId = ++requestCounter.value;
    currentRequestId.value = requestId;
    isThinking.value = true;

    aiWorker.value.postMessage({
      requestId,
      board: gameStore.board,
      player: gameStore.aiColor,
      difficulty: gameStore.aiDifficulty,
      boardSize: gameStore.boardSize,
      timeLimit: getAITimeLimit(gameStore.aiDifficulty, gameStore.boardSize)
    });
  }

  return { isThinking, initAI, makeMove, cleanup };
}
```

**AI Worker 实现**:
```typescript
// frontend/src/workers/ai-worker.ts

import { GomokuAI } from './ai';

let ai: GomokuAI | null = null;

self.onmessage = (e) => {
  const { requestId, board, player, difficulty, boardSize, timeLimit } = e.data;

  if (!ai || ai.size !== boardSize || ai.difficulty !== difficulty) {
    ai = new GomokuAI(difficulty, boardSize);
  }

  const startTime = Date.now();
  const move = ai.getBestMove(board, player, timeLimit);
  const thinkTime = Date.now() - startTime;

  self.postMessage({
    requestId,
    success: !!move,
    move,
    thinkTime
  });
};
```

**WebSocket 消息处理**:
```typescript
// frontend/src/App.vue

function setupWSMessageHandler() {
  wsStore.onMessage((data: any) => {
    switch (data.type) {
      case 'room_created':
        gameStore.roomCode = data.roomCode;
        gameStore.myColor = 1; // 房主是黑棋
        gameStore.isHost = true;
        gameStore.myUserId = data.userId;
        wsStore.saveRoomInfo(data.roomCode, gameStore.myName, data.userId, true, 1);
        panels.value.createRoom = false;
        panels.value.waiting = true;
        break;

      case 'room_joined':
        gameStore.roomCode = data.roomCode;
        gameStore.myColor = data.color;
        gameStore.myUserId = data.userId;
        gameStore.isHost = false;
        wsStore.saveRoomInfo(data.roomCode, gameStore.myName, data.userId, false, data.color);

        if (data.opponent) {
          // 房间已满，直接开始游戏
          gameStore.opponentName = data.opponent.name;
          gameStore.undoLimit = data.undoLimit || 3;
          gameStore.startGame('online', {
            player1: gameStore.myColor === 1 ? gameStore.myName : data.opponent.name,
            player2: gameStore.myColor === 2 ? gameStore.myName : data.opponent.name,
            time: data.time || 300,
            size: data.size || 15,
            matchMode: (data.matchMode || 1) as MatchMode
          });
          currentScreen.value = 'game';
        } else {
          panels.value.waiting = true;
        }
        break;

      case 'opponent_joined':
        gameStore.opponentName = data.opponent.name;
        panels.value.waiting = false;
        gameStore.undoLimit = data.undoLimit || 3;
        gameStore.startGame('online', {
          player1: gameStore.myColor === 1 ? gameStore.myName : data.opponent.name,
          player2: gameStore.myColor === 2 ? gameStore.myName : data.opponent.name,
          time: data.time || 300,
          size: data.size || 15,
          matchMode: (data.matchMode || 1) as MatchMode
        });
        currentScreen.value = 'game';
        break;

      case 'piece_placed':
        gameStore.doPlacePiece(data.row, data.col, data.player);
        if (data.currentPlayer) {
          gameStore.currentPlayer = data.currentPlayer;
        }
        if (data.players) {
          data.players.forEach((p: any) => {
            if (p) {
              gameStore.players[p.color] = {
                name: p.name,
                time: p.time,
                moves: p.moves,
                undoLeft: p.undoLeft
              };
            }
          });
        }
        break;

      case 'time_sync':
        // 服务器心跳同步时间
        if (data.players) {
          data.players.forEach((p: any) => {
            if (p) {
              gameStore.players[p.color].time = p.time;
            }
          });
        }
        if (data.currentPlayer) {
          gameStore.currentPlayer = data.currentPlayer;
        }
        break;

      case 'game_over':
        if (data.matchWins) {
          gameStore.matchWins = data.matchWins;
        }
        if (data.winningMove) {
          gameStore.doPlacePiece(data.winningMove.row, data.winningMove.col, data.winner);
        }
        gameStore.endGame(data.winner || 0);
        showWinModal.value = true;

        // 检查比赛是否结束
        const targetWins = Math.ceil(gameStore.matchMode / 2);
        if (gameStore.matchWins[1] >= targetWins || gameStore.matchWins[2] >= targetWins) {
          gameStore.matchEnded = true;
          wsStore.clearRoomInfo();
        }
        break;

      case 'play_again':
        // 双方都准备好后，服务器发送 play_again 消息
        if (data.players) {
          const myPlayer = data.players.find((p: any) => p && p.id === gameStore.myUserId);
          if (myPlayer) {
            gameStore.myColor = myPlayer.color;
            gameStore.myPlayerIndex = data.players.indexOf(myPlayer);
            const opponent = data.players.find((p: any) => p && p.color !== myPlayer.color);
            if (opponent) {
              gameStore.opponentName = opponent.name;
            }
          }

        if (data.currentPlayer) gameStore.currentPlayer = data.currentPlayer;

                name: p.name,
                time: p.time,
```

```typescript
// frontend/src/stores/game.ts

function checkWin(row: number, col: number, player: Player): WinResult[] | null {
  const directions: [number, number][][] = [
    [[0, 1], [0, -1]],   // 水平
    [[1, 0], [-1, 0]],   // 垂直
    [[1, 1], [-1, -1]],  // 主对角线
    [[1, -1], [-1, 1]]   // 副对角线
  ];

  for (const dir of directions) {
    let count = 1;
    const line: WinResult[] = [{ row, col }];

    // 检查两个方向
    for (const [dr, dc] of dir) {
      let r = row + dr, c = col + dc;
      while (
        r >= 0 && r < boardSize.value &&
        c >= 0 && c < boardSize.value &&
        board.value[r][c] === player
      ) {
        count++;
        line.push({ row: r, col: c });
        r += dr;
        c += dc;
      }
    }

    if (count >= 5) return line;
  }

  return null;
}
```

```javascript
function connectWebSocket() {
    return new Promise((resolve, reject) => {
        if (ws && ws.readyState === WebSocket.OPEN) {
            resolve();
            return;
        }

        try {
            ws = new WebSocket(WS_SERVER);

            ws.onopen = () => {
                wsConnected = true;
                reconnectAttempts = 0;
                document.getElementById('reconnectOverlay').style.display = 'none';
                resolve();

                if (GameState.gameMode === 'online' && GameState.isPlaying && 
                    reconnectAttempts < MAX_RECONNECT) {
                    showReconnect();
                }
            };

            ws.onerror = (err) => {
                wsConnected = false;
                reject(err);
            };

            ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                handleServerMessage(data);
            };

            setTimeout(() => { 
                if (!wsConnected) reject(new Error('Connection timeout')); 
            }, 10000);
        } catch (err) {
            reject(err);
        }
    });
}
```

```javascript
function handlePlayerDisconnect(ws) {
    const client = clients.get(ws);
    if (!client) return;

    const { roomCode, color } = client;
    const room = rooms.get(roomCode);

    if (!room) {
        clients.delete(ws);
        return;
    }

    const playerIndex = room.players.findIndex(p => p && p.ws === ws);
    if (playerIndex === -1) {
        clients.delete(ws);
        return;
    }

    const player = room.players[playerIndex];
    if (player) {
        player.ws = null;
        player.disconnectedAt = Date.now();
    }

    clients.delete(ws);

    // 如果游戏正在进行，通知对手
    if (room.status === 'playing') {
        const opponent = room.players.find(p => p && p.ws && p.ws !== ws);

        if (opponent && opponent.ws && opponent.ws.readyState === WebSocket.OPEN) {
            send(opponent.ws, {
                type: 'opponent_disconnected',
                message: '对手已掉线，等待重连...',
                reconnectTimeout: RECONNECT_TIMEOUT
            });
        }

        // 设置重连超时（60秒）
        room.reconnectTimer = setTimeout(() => {
            if (room.players[playerIndex] && !room.players[playerIndex].ws) {
                const disconnectedPlayer = room.players[playerIndex];
                const isHost = disconnectedPlayer && disconnectedPlayer.id === room.hostId;

                if (isHost) {
                    room.matchWins[2]++;
                } else {
                    room.matchWins[1]++;
                }

                const targetWins = Math.ceil(room.matchMode / 2);
                const matchEnded = room.matchWins[1] >= targetWins || 
                                   room.matchWins[2] >= targetWins;

                if (matchEnded) {
                    room.status = 'finished';
                    room.finishedAt = Date.now();
                } else {
                    room.status = 'finished';
                    room.finishedAt = Date.now();
                }

                room.playAgainRequested = false;

                if (opponent && opponent.ws && opponent.ws.readyState === WebSocket.OPEN) {
                    const winner = isHost ? 2 : 1;
                    send(opponent.ws, {
                        type: 'game_over',
                        winner: winner,
                        reason: 'disconnect',
                        matchWins: room.matchWins,
                        matchEnded: matchEnded
                    });
                }
            }
        }, RECONNECT_TIMEOUT);
    }
    
    // 如果是等待中的房间，房主掉线则删除房间
    if (room.status === 'waiting' && color === 1) {
        rooms.delete(roomCode);
    }
```

```javascript
checkOpeningBook(board, player) {
    const totalPieces = this.countPieces(board);
    
    // 空棋盘，下天元
    if (totalPieces === 0) {
        return this.openingBook.empty;
    }
    
    // 只有1颗棋子（对手下了天元）
    if (totalPieces === 1) {
        const center = Math.floor(this.size / 2);
        if (board[center][center] !== 0) {
            const responses = this.openingBook.center_response[this.size] || 
                              this.openingBook.center_response[15];
            return responses[Math.floor(Math.random() * responses.length)];
        }
    }
    
    // 开局阶段（前10步），尝试找定式，但要先检查威胁
    if (totalPieces < 10) {
        const opponent = 3 - player;
        
        // 优先检查对手的威胁（活四、冲四、活三）
        const candidates = this.getCandidateMoves(board);
        for (const move of candidates) {
            board[move.row][move.col] = opponent;
            const opponentScore = this.evaluatePoint(board, move.row, move.col, opponent, false);
            board[move.row][move.col] = 0;
            
            // 如果对手有活四或冲四，必须防守
            if (opponentScore >= this.patternScores.LIVE_FOUR) {
                console.log('AI: 开局检测到对手活四，紧急防守', move);
                return move;
            }
            // 如果对手有活三，也需要防守
            if (opponentScore >= this.patternScores.LIVE_THREE) {
                console.log('AI: 开局检测到对手活三，防守', move);
                return move;
            }
        }
        
        // 如果没有威胁，尝试使用定式
        const bookMove = this.findBookMove(board, player);
```

#### 2. 候选点生成

```javascript
getCandidateMoves(board) {
    const moves = [];
    const visited = new Set();
    
    const pieceCount = this.searchStats?.pieceCount || this.countPieces(board);
    const complexity = pieceCount / (this.size * this.size);
    
    // 根据棋盘大小和复杂度动态调整参数
    const isEndgame = complexity > 0.75;
    const maxMoves = this.size === 19 ? (isEndgame ? 100 : 80) : (isEndgame ? 80 : 60);
    const searchDistance = isEndgame ? 3 : 2; // 终局扩大搜索范围
    
    for (let i = 0; i < this.size; i++) {
        for (let j = 0; j < this.size; j++) {
            if (board[i][j] !== 0) {
                // 检查周围的空位
                for (let di = -searchDistance; di <= searchDistance; di++) {
                    for (let dj = -searchDistance; dj <= searchDistance; dj++) {
                        const distance = isEndgame 
                            ? Math.sqrt(di*di + dj*dj) 
                            : Math.abs(di) + Math.abs(dj);

                        if (distance > searchDistance) continue;
                        
                        const ni = i + di;
                        const nj = j + dj;
                        const key = ni * this.size + nj;
                        
                        if (ni >= 0 && ni < this.size && nj >= 0 && nj < this.size &&
                            board[ni][nj] === 0 && !visited.has(key)) {
                            visited.add(key);
                            moves.push({ row: ni, col: nj });
                            
                            if (moves.length >= maxMoves) {
                                break;
                            }
                        }
                    }
                    if (moves.length >= maxMoves) break;
                }
                if (moves.length >= maxMoves) break;
            }
            if (moves.length >= maxMoves) break;
        }
    }
    
    // 如果棋盘为空，返回中心点
    if (moves.length === 0) {
        const center = Math.floor(this.size / 2);
        moves.push({ row: center, col: center });
    }
    
    return moves;
}
```

#### 3. 走法评估

```javascript
evaluateMoves(moves, board, player) {
    const opponent = 3 - player;
    let highDefenseMoves = [];

    // 获取动态防守权重
    const dynamicDefenseMultiplier = this.getDynamicDefenseMultiplier(board, player);

    for (const move of moves) {
        // 评估进攻收益
        board[move.row][move.col] = player;
        const attackScore = this.evaluatePoint(board, move.row, move.col, player, true);
        board[move.row][move.col] = 0;

        // 评估防守威胁
        board[move.row][move.col] = opponent;
        const defenseScore = this.evaluatePoint(board, move.row, move.col, opponent, false);
        board[move.row][move.col] = 0;
        
        // 基础分数：进攻 - 防守×权重
        move.score = attackScore - (defenseScore * dynamicDefenseMultiplier);
        move.attackScore = attackScore;
        move.defenseScore = defenseScore;
        
        // 对高威胁棋型给予额外防守权重
        if (defenseScore >= this.patternScores.LIVE_FOUR) {
            move.score += 300000;  // 强制防守
            highDefenseMoves.push({ ...move, reason: 'LIVE_FOUR' });
        }
        else if (defenseScore >= this.patternScores.RUSH_FOUR) {
            move.score += 80000;
            highDefenseMoves.push({ ...move, reason: 'RUSH_FOUR' });
        }
        else if (defenseScore >= this.patternScores.LIVE_THREE) {
            move.score += 15000;
            highDefenseMoves.push({ ...move, reason: 'LIVE_THREE' });
        }
        
        // 加上位置权重
        move.score += this.positionWeight[move.row][move.col];
    }
    
    // 按分数降序排序
    moves.sort((a, b) => b.score - a.score);

    // 只保留前 30 个最有希望的走法
    if (moves.length > 30) {
        return moves.slice(0, 30);
    }
    return moves;
}
```

#### 4. Alpha-Beta 剪枝搜索

```javascript
pvs(board, depth, alpha, beta, player, hash) {
    this.nodeCount++;
    this.searchStats.nodes++;

    // 时间检查
    if (this.nodeCount % 16 === 0) {
        if (Date.now() - this.startTime > this.timeLimit) {
            return alpha;
        }
    }
    
    // 检查胜负
    const opponent = 3 - player;
    if (this.checkWin(board, opponent)) {
        return -this.patternScores.FIVE + (100 - depth);
    }
    
    // 叶节点评估
    if (depth <= 0) {
        this.searchStats.evaluations++;
        return this.evaluate(board, player);
    }
    
    // 置换表查询
    const hashStr = hash.toString();
    const cached = this.transpositionTable.get(hashStr);
    
    const currentPieceCount = this.searchStats.pieceCount || this.countPieces(board);
    
    if (cached && cached.depth >= depth && cached.age === this.tableAge && 
        cached.pieceCount === currentPieceCount) {
        this.searchStats.cacheHits++;
        if (cached.flag === 'exact') {
            return cached.score;
        } else if (cached.flag === 'lower') {
            alpha = Math.max(alpha, cached.score);
        } else if (cached.flag === 'upper') {
            beta = Math.min(beta, cached.score);
        }
        if (alpha >= beta) {
            return cached.score;
        }
    } else {
        this.searchStats.cacheMisses++;
    }
    
    // 获取候选走法
    const moves = this.getCandidateMoves(board);
    
    if (moves.length === 0) {
        return 0;
    }
    
    // 排序走法
    this.sortMoves(moves);
    
    let bestScore = -Infinity;
    let flag = 'upper';
    
    for (let i = 0; i < moves.length; i++) {
        const move = moves[i];
        
        // 执行走法
        board[move.row][move.col] = player;
        const newHash = hash ^ this.zobrist[move.row][move.col][player];
        
        let score;
        if (i === 0) {
            score = -this.pvs(board, depth - 1, -beta, -alpha, 3 - player, newHash);
        } else {
            // 零窗口搜索
            score = -this.pvs(board, depth - 1, -alpha - 1, -alpha, 3 - player, newHash);
            if (score > alpha && score < beta) {
                // 重新搜索
                score = -this.pvs(board, depth - 1, -beta, -score, 3 - player, newHash);
            }
        }
        
        // 撤销走法
        board[move.row][move.col] = 0;
        
        if (score > bestScore) {
            bestScore = score;
        }
        
        if (score > alpha) {
            alpha = score;
            flag = 'exact';
            this.historyTable[move.row][move.col] += depth * depth;
        }
        
        if (alpha >= beta) {
            flag = 'lower';
            this.searchStats.cutoffs++;
            break;
        }
    }
    
    // 存入置换表
    this.transpositionTable.set(hashStr, {
      score: bestScore,
      depth: depth,
      flag: flag,
      age: this.tableAge,
      pieceCount: currentPieceCount
    });
    
    return bestScore;
}
```

#### 5. 获取最佳走法

```javascript
getBestMove(board, player, timeLimit = null) {
    // 输入验证
    if (!board || !Array.isArray(board) || board.length === 0) {
        throw new Error('Invalid board: board is null or empty');
    }
    if (player !== 1 && player !== 2) {
        throw new Error('Invalid player: must be 1 or 2');
    }
    if (board.length !== this.size || board[0].length !== this.size) {
        throw new Error(`Invalid board size: expected ${this.size}x${this.size}`);
    }
    
    const pieceCount = this.countPieces(board);
    const complexity = pieceCount / (this.size * this.size);
    
    if (pieceCount >= this.size * this.size) {
        console.log('AI: 棋盘已满，无法落子');
        return null;
    }
    
    // 初始化搜索统计
    this.searchStats = {
        nodes: 0,
        cutoffs: 0,
        cacheHits: 0,
        cacheMisses: 0,
        evaluations: 0,
        pieceCount: pieceCount,
        complexity: complexity
    };
    
    // 动态调整时间限制
    let adjustedTimeLimit = timeLimit || this.timeLimitConfig || 1000;
    
    if (complexity < 0.1) {
        // 开局阶段
    } else if (complexity < 0.25) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.3);
    } else if (complexity < 0.6) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 0.8);
    } else if (complexity < 0.8) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.3);
    } else {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.5);
    }
    
    this.timeLimit = adjustedTimeLimit;
    this.startTime = Date.now();
    this.nodeCount = 0;
    this.tableAge++;
    
    // 清理过旧的置换表项
    if (this.transpositionTable.size > this.maxTableSize) {
        this.cleanTranspositionTable();
    }
    
    // 重置历史启发表
    this.historyTable = this.createArray2D(0);
    
    // 检查开局库
    const bookMove = this.checkOpeningBook(board, player);
    if (bookMove) {
        console.log('AI: 使用开局库', bookMove);
        return bookMove;
    }
    
    // 终局优化
    if (complexity > 0.7) {
        this.depthConfig.max += 1;
        console.log('AI: 终局阶段，增加搜索深度');
    }

    // 终局多重威胁检测
    if (complexity > 0.7) {
        const multipleThreats = this.detectMultipleThreats(board, player);
        if (multipleThreats.length > 0) {
            const winningThreat = multipleThreats.find(t =>
                t.attackScore >= this.patternScores.LIVE_FOUR || 
                t.attackScore >= this.patternScores.RUSH_FOUR
            );
            if (winningThreat) {
                console.log('AI: 终局找到必胜走法', winningThreat);
                return winningThreat;
            }
            const defenseThreat = multipleThreats.find(t => 
                t.defenseScore >= this.patternScores.LIVE_FOUR
            );
            if (defenseThreat) {
                console.log('AI: 终局必须防守', defenseThreat);
                return defenseThreat;
            }
        }
    }

    let bestMove = null;
    let bestScore = -Infinity;
    
    // 迭代加深搜索
    for (let depth = 2; depth <= this.depthConfig.max; depth += 2) {
        const elapsedTime = Date.now() - this.startTime;
        const remainingTime = this.timeLimit - elapsedTime;
        
        if (remainingTime < this.timeLimit * 0.2) {
            console.log('AI: 时间不足，停止迭代加深');
            break;
        }
        
        if (elapsedTime > this.timeLimit * 0.8) {
            console.log('AI: 时间已用80%，停止迭代加深');
            break;
        }
        
        const result = this.searchRoot(board, player, depth);
        
        if (result && result.move) {
            bestMove = result.move;
            bestScore = result.score;
            
            if (bestScore >= this.patternScores.FIVE) {
                break;
            }
        }
        
        if (depth >= this.depthConfig.max) {
            break;
        }
    }
    
    // 如果没有找到走法，下在中心
    if (!bestMove) {
        const center = Math.floor(this.size / 2);
        if (board[center][center] === 0) {
            bestMove = { row: center, col: center };
        } else {
            for (let i = 0; i < this.size; i++) {
                for (let j = 0; j < this.size; j++) {
                    if (board[i][j] === 0) {
                        bestMove = { row: i, col: j };
                        break;
                    }
                }
                if (bestMove) break;
            }
        }
    }
    
    const searchTime = Date.now() - this.startTime;
    console.log('AI: 搜索完成', {
        move: bestMove,
        time: `${searchTime}ms`,
        stats: {
            nodes: this.searchStats.nodes,
            cutoffs: this.searchStats.cutoffs,
            cacheHits: this.searchStats.cacheHits,
            cacheMisses: this.searchStats.cacheMisses,
            evaluations: this.searchStats.evaluations
        }
    });
    
    return bestMove;
}
```

## 核心算法总结

### AI 算法核心特点

1. **Alpha-Beta 剪枝 + PVS（主变搜索）**
   - 使用零窗口搜索提高效率
   - 迭代加深搜索

2. **Zobrist 置换表**
   - 使用 LRU 缓存存储搜索结果
   - 避免重复计算相同局面
   - 最大容量 50MB

3. **历史启发**
   - 记录走法历史，优先搜索好的走法
   - 提高剪枝效率

4. **候选点剪枝**
   - 只搜索已有棋子周围 2-3 格内的空位
   - 终局阶段扩大搜索范围

5. **开局库**
   - 支持花月、浦月等定式
   - 前 10 步使用定式走法

6. **动态时间调整**
   - 开局和终局给更多时间
   - 中局减少时间

7. **动态防守权重**
   - 根据局势调整防守权重
   - 优势时加强进攻，劣势时加强防守

8. **终局优化**
   - 增加搜索深度
   - 检测多重威胁
