## 项目概述

本项目是一个功能完整的五子棋在线对战平台，采用前后端分离架构：

- **前端**: 纯 HTML/CSS/JavaScript 单页应用（index.html）
- **后端**: Node.js + WebSocket 服务器（server/server.js）
- **AI 引擎**: 基于 Alpha-Beta 剪枝算法（ai.js）
- **AI Worker**: Web Worker 包装器（ai-worker.js）

## 游戏模式详解

### 1. 本地对战模式

**启动流程**:
```
菜单 → 点击"本地对战"按钮 → showLocalSetup() → 显示设置面板
  ↓
用户填写设置（玩家名称、对局时长、棋盘大小、悔棋次数）
  ↓
点击"开始游戏" → startLocalGame() → initGame('local', config)
```

**核心代码**:
```javascript
function startLocalGame() {
    const player1 = validateNickname(document.getElementById('localPlayer1').value) || '玩家1';
    const player2 = validateNickname(document.getElementById('localPlayer2').value) || '玩家2';

```

**特点**:
- 无需网络连接
- 两人在同一设备上轮流落子
- 直接悔棋（无需对方同意）
- 悔棋次数有限制（1/3/5/无限）
- 支持计时系统
- 不支持比赛模式（始终单局决胜）

### 2. 人机对战模式

**启动流程**:
```
菜单 → 点击"人机对战"按钮 → showAISetup() → 显示设置面板
  ↓
用户填写设置（昵称、执子颜色、AI难度、棋盘大小、对局时长）
  ↓
点击"开始游戏" → startAIGame() → 创建 AI 实例 → initGame('ai', config)
  ↓
如果 AI 执黑先手 → setTimeout(makeAIMove, 300)
```

**核心代码**:
```javascript
function startAIGame() {
    const playerName = validateNickname(document.getElementById('aiPlayerName').value) || '玩家';
    const playerColor = parseInt(document.getElementById('aiPlayerColor').value);
    const difficulty = document.getElementById('aiDifficulty').value;
    const size = parseInt(document.getElementById('aiBoardSize').value);
    const time = parseInt(document.getElementById('aiGameTime').value);

    // 初始化 AI 引擎
    aiEngine = new GomokuAI(difficulty, size);

    GameState.undoLimit = 999; // AI 模式允许无限悔棋
    GameState.matchMode = 1;
    GameState.myColor = playerColor;
    GameState.myName = playerName;
    GameState.opponentName = 'AI';
    GameState.aiColor = playerColor === 1 ? 2 : 1;
    GameState.aiDifficulty = difficulty;

    const blackName = playerColor === 1 ? playerName : 'AI';
    const whiteName = playerColor === 2 ? playerName : 'AI';

    initGame('ai', { player1: blackName, player2: whiteName, time, size });

    // 如果 AI 执黑（先手），让 AI 先下
    if (GameState.aiColor === 1) {
        setTimeout(makeAIMove, 300);
    }
}
```

**AI 思考流程**:
```
轮到 AI 落子 → makeAIMove()
  ↓
生成唯一请求 ID → ++aiRequestCounter
  ↓
使用 Web Worker 发送计算请求
  ↓
Worker 返回结果 → onmessage 回调
  ↓
检查请求 ID 是否匹配（忽略过期响应）
  ↓
扣除 AI 思考时间
  ↓
执行落子 → doPlacePiece()
```

**AI Web Worker 通信**:
```javascript
// 发送请求
GameState.aiWorker.postMessage({
    requestId: requestId,
    board: GameState.board,
    player: GameState.aiColor,
    difficulty: GameState.aiDifficulty,
    boardSize: GameState.boardSize,
    timeLimit: timeLimit
});

// 接收响应
GameState.aiWorker.onmessage = function(e) {
    const result = e.data;
    const dataRequestId = result.requestId;

    // 只处理当前请求的响应
    if (dataRequestId !== currentAiRequestId) {
        console.log('AI: 忽略过期响应');
        return;
    }

    if (result.success && result.move) {
        // 计算 AI 思考耗时
        const thinkTime = Math.ceil(result.thinkTime / 1000);

        // 扣除 AI 的思考时间
        if (thinkTime > 0 && GameState.players[GameState.aiColor]) {
            GameState.players[GameState.aiColor].time = Math.max(0, 
                GameState.players[GameState.aiColor].time - thinkTime);
            
            // 检查 AI 是否超时
            if (GameState.players[GameState.aiColor].time <= 0) {
                const winner = GameState.aiColor === 1 ? 2 : 1;
                endGame(winner);
                return;
            }
        }

        doPlacePiece(result.move.row, result.move.col, GameState.aiColor);
    }
};
```

**特点**:
- 使用 Web Worker 后台计算，避免阻塞 UI
- 支持三种难度（简单/中等/困难）
- 允许无限悔棋
- 悔棋时撤销两步（玩家 + AI）
- AI 思考时间计入对局时间
- 支持 13×13、15×15、19×19 三种棋盘

### 悔棋按钮详细逻辑

```javascript
function requestUndo() {
    // 游戏结束后不允许悔棋
    if (!GameState.isPlaying) {
        showToast('游戏已结束，无法悔棋');
        return;
    }

    if (GameState.moveHistory.length === 0) {
        showToast('没有可以悔棋的步骤');
        return;
    }

    const myColor = GameState.myColor;
    if (GameState.players[myColor].undoLeft <= 0 && GameState.gameMode !== 'local') {
        showToast('悔棋次数已用完');
        return;
    }

    const lastMove = GameState.moveHistory[GameState.moveHistory.length - 1];

    if (GameState.gameMode === 'online') {
        // 在线模式：只能悔自己刚下的棋（轮到对手时才能请求）
        // lastMove.player === myColor 表示我刚下完棋，现在轮到对手，可以请求悔棋
        if (lastMove.player !== myColor) {
            showToast('只能悔自己刚下的棋，请等对手下完后再请求');
            return;
        }

        if (GameState.undoRequested) {
            showToast('已有待处理的悔棋请求');
            return;
        }
        sendToServer({ type: 'undo_request' });
        showToast('已发送悔棋请求');
    } else {
        // 本地模式：悔掉最后一步棋
        doUndo();
    }
}
```

**悔棋执行逻辑**:
```javascript
function doUndo() {
    if (GameState.moveHistory.length === 0) return;

    // AI 模式：悔两步（玩家 + AI）
    if (GameState.gameMode === 'ai') {
        if (GameState.moveHistory.length >= 2) {
            // 撤销 AI 的棋（如果最后一步是 AI 下的）
            const lastMove = GameState.moveHistory[GameState.moveHistory.length - 1];
            if (lastMove.player === GameState.aiColor) {
                const aiMove = GameState.moveHistory.pop();
                GameState.board[aiMove.row][aiMove.col] = 0;
                GameState.players[aiMove.player].moves--;
                // 移除棋子显示...
            }
        }

        // 撤销玩家的棋
        if (GameState.moveHistory.length >= 1) {
            const lastMove = GameState.moveHistory[GameState.moveHistory.length - 1];
            if (lastMove.player !== GameState.aiColor) {
                const playerMove = GameState.moveHistory.pop();
                GameState.board[playerMove.row][playerMove.col] = 0;
                GameState.players[playerMove.player].moves--;
                // 移除棋子显示...
                GameState.currentPlayer = lastMove.player;
            }
        }
    } else {
        // 本地/在线模式：撤销最后一步
        const lastMove = GameState.moveHistory.pop();
        if (lastMove) {
            GameState.board[lastMove.row][lastMove.col] = 0;
            GameState.players[lastMove.player].moves--;
            // 移除棋子显示...
            GameState.currentPlayer = lastMove.player;
        }

        // 本地模式：消耗悔棋次数
        if (GameState.gameMode === 'local') {
            GameState.players[lastMove.player].undoLeft--;
        }
    }

    clearPreview();
    updateUI();
}
```

```javascript
    const readyBtn = document.getElementById('readyBtn');
    const restartBtn = document.getElementById('restartBtn');
    const boardReadyStatus = document.getElementById('boardReadyStatus');

    if (readyBtn) readyBtn.style.display = 'none';
    if (restartBtn) restartBtn.style.display = 'inline-block';
    if (boardReadyStatus) boardReadyStatus.style.display = 'none';

    // 更新局数和比分
    if (data.currentRound) {
        GameState.currentRound = data.currentRound;
    }
    if (data.matchWins) {
        GameState.matchWins = data.matchWins;
    }

    // 更新玩家数据
    if (data.players) {
        const myPlayer = data.players.find(p => p && p.id === GameState.myUserId);
        if (myPlayer) {
            GameState.myColor = myPlayer.color;
            GameState.myPlayerIndex = data.players.indexOf(myPlayer);
            const opponent = data.players.find(p => p && p.color !== myPlayer.color);
            if (opponent) {
                GameState.opponentName = opponent.name;
            }
        }

        data.players.forEach(p => {
            if (p) {
                GameState.players[p.color] = {
                    name: p.name,
                    time: p.time,
                    moves: p.moves,
                    undoLeft: p.undoLeft
                };
            }
        });
    }

    restartGame();
}
```

## WebSocket 操作详解

### WebSocket 连接

**连接函数**:
```javascript
function connectWebSocket() {
    return new Promise((resolve, reject) => {
        if (ws && ws.readyState === WebSocket.OPEN) {
            resolve();
            return;
        }

        try {
            ws = new WebSocket(WS_SERVER);

            ws.onopen = () => {
                wsConnected = true;
                reconnectAttempts = 0;
                document.getElementById('reconnectOverlay').style.display = 'none';
                resolve();
            };

            ws.onclose = () => {
                wsConnected = false;
                if (GameState.gameMode === 'online' && GameState.isPlaying && 
                    reconnectAttempts < MAX_RECONNECT) {
                    showReconnect();
                }
            };

            ws.onerror = (err) => {
                wsConnected = false;
                reject(err);
            };

            ws.onmessage = (event) => {
                const data = JSON.parse(event.data);
                handleServerMessage(data);
            };

            setTimeout(() => { 
                if (!wsConnected) reject(new Error('Connection timeout')); 
            }, 10000);
        } catch (err) {
            reject(err);
        }
    });
}
```

**发送消息**:
```javascript
function sendToServer(data) {
    if (!ws || ws.readyState !== WebSocket.OPEN) {
        console.error('WebSocket 未连接，无法发送消息', data);
        return false;
    }
    ws.send(JSON.stringify(data));
    return true;
}
```

```javascript
checkOpeningBook(board, player) {
    const totalPieces = this.countPieces(board);
    
    // 空棋盘，下天元
    if (totalPieces === 0) {
        return this.openingBook.empty;
    }
    
    // 只有1颗棋子（对手下了天元）
    if (totalPieces === 1) {
        const center = Math.floor(this.size / 2);
        if (board[center][center] !== 0) {
            const responses = this.openingBook.center_response[this.size] || 
                              this.openingBook.center_response[15];
            return responses[Math.floor(Math.random() * responses.length)];
        }
    }
    
    // 开局阶段（前10步），尝试找定式，但要先检查威胁
    if (totalPieces < 10) {
        const opponent = 3 - player;
        
        // 优先检查对手的威胁（活四、冲四、活三）
        const candidates = this.getCandidateMoves(board);
        for (const move of candidates) {
            board[move.row][move.col] = opponent;
            const opponentScore = this.evaluatePoint(board, move.row, move.col, opponent, false);
            board[move.row][move.col] = 0;
            
            // 如果对手有活四或冲四，必须防守
            if (opponentScore >= this.patternScores.LIVE_FOUR) {
                console.log('AI: 开局检测到对手活四，紧急防守', move);
                return move;
            }
            // 如果对手有活三，也需要防守
            if (opponentScore >= this.patternScores.LIVE_THREE) {
                console.log('AI: 开局检测到对手活三，防守', move);
                return move;
```

#### 2. 候选点生成

```javascript
getCandidateMoves(board) {
    const moves = [];
    const visited = new Set();
    
    const pieceCount = this.searchStats?.pieceCount || this.countPieces(board);
    const complexity = pieceCount / (this.size * this.size);
    
    // 根据棋盘大小和复杂度动态调整参数
    const isEndgame = complexity > 0.75;
    const maxMoves = this.size === 19 ? (isEndgame ? 100 : 80) : (isEndgame ? 80 : 60);
    const searchDistance = isEndgame ? 3 : 2; // 终局扩大搜索范围
    
    for (let i = 0; i < this.size; i++) {
        for (let j = 0; j < this.size; j++) {
            if (board[i][j] !== 0) {
                // 检查周围的空位
                for (let di = -searchDistance; di <= searchDistance; di++) {
                    for (let dj = -searchDistance; dj <= searchDistance; dj++) {
                        const distance = isEndgame 
                            ? Math.sqrt(di*di + dj*dj) 
                            : Math.abs(di) + Math.abs(dj);

                        if (distance > searchDistance) continue;
                        
                        const ni = i + di;
                        const nj = j + dj;
                        const key = ni * this.size + nj;
                        
                        if (ni >= 0 && ni < this.size && nj >= 0 && nj < this.size &&
                            board[ni][nj] === 0 && !visited.has(key)) {
                            visited.add(key);
                            moves.push({ row: ni, col: nj });
                            
                            if (moves.length >= maxMoves) {
                                break;
                            }
                        }
                    }
                    if (moves.length >= maxMoves) break;
                }
                if (moves.length >= maxMoves) break;
            }
            if (moves.length >= maxMoves) break;
        }
    }
    
    // 如果棋盘为空，返回中心点
    if (moves.length === 0) {
        const center = Math.floor(this.size / 2);
        moves.push({ row: center, col: center });
    }
    
    return moves;
}
```

#### 3. 走法评估

```javascript
evaluateMoves(moves, board, player) {
    const opponent = 3 - player;
    let highDefenseMoves = [];

    // 获取动态防守权重
    const dynamicDefenseMultiplier = this.getDynamicDefenseMultiplier(board, player);

    for (const move of moves) {
        // 评估进攻收益
        board[move.row][move.col] = player;
        const attackScore = this.evaluatePoint(board, move.row, move.col, player, true);
        board[move.row][move.col] = 0;

        // 评估防守威胁
        board[move.row][move.col] = opponent;
        const defenseScore = this.evaluatePoint(board, move.row, move.col, opponent, false);
        board[move.row][move.col] = 0;
        
        // 基础分数：进攻 - 防守×权重
        move.score = attackScore - (defenseScore * dynamicDefenseMultiplier);
        move.attackScore = attackScore;
        move.defenseScore = defenseScore;
        
        // 对高威胁棋型给予额外防守权重
        if (defenseScore >= this.patternScores.LIVE_FOUR) {
            move.score += 300000;  // 强制防守
            highDefenseMoves.push({ ...move, reason: 'LIVE_FOUR' });
        }
        else if (defenseScore >= this.patternScores.RUSH_FOUR) {
            move.score += 80000;
            highDefenseMoves.push({ ...move, reason: 'RUSH_FOUR' });
        }
        else if (defenseScore >= this.patternScores.LIVE_THREE) {
            move.score += 15000;
            highDefenseMoves.push({ ...move, reason: 'LIVE_THREE' });
        }
        
        // 加上位置权重
        move.score += this.positionWeight[move.row][move.col];
    }
    
    // 按分数降序排序
    moves.sort((a, b) => b.score - a.score);

    // 只保留前 30 个最有希望的走法
    if (moves.length > 30) {
        return moves.slice(0, 30);
    }
    return moves;
}
```

#### 4. Alpha-Beta 剪枝搜索

```javascript
pvs(board, depth, alpha, beta, player, hash) {
    this.nodeCount++;
    this.searchStats.nodes++;

    // 时间检查
    if (this.nodeCount % 16 === 0) {
        if (Date.now() - this.startTime > this.timeLimit) {
            return alpha;
        }
    }
    
    // 检查胜负
    const opponent = 3 - player;
    if (this.checkWin(board, opponent)) {
        return -this.patternScores.FIVE + (100 - depth);
    }
    
    // 叶节点评估
    if (depth <= 0) {
        this.searchStats.evaluations++;
        return this.evaluate(board, player);
    }
    
    // 置换表查询
    const hashStr = hash.toString();
    const cached = this.transpositionTable.get(hashStr);
    
    const currentPieceCount = this.searchStats.pieceCount || this.countPieces(board);
    
    if (cached && cached.depth >= depth && cached.age === this.tableAge && 
        cached.pieceCount === currentPieceCount) {
        this.searchStats.cacheHits++;
        if (cached.flag === 'exact') {
            return cached.score;
        } else if (cached.flag === 'lower') {
            alpha = Math.max(alpha, cached.score);
        } else if (cached.flag === 'upper') {
            beta = Math.min(beta, cached.score);
        }
        if (alpha >= beta) {
            return cached.score;
        }
    } else {
        this.searchStats.cacheMisses++;
    }
    
    // 获取候选走法
    const moves = this.getCandidateMoves(board);
    
    if (moves.length === 0) {
        return 0;
    }
    
    // 排序走法
    this.sortMoves(moves);
    
    let bestScore = -Infinity;
    let flag = 'upper';
    
    for (let i = 0; i < moves.length; i++) {
        const move = moves[i];
        
        // 执行走法
        board[move.row][move.col] = player;
        const newHash = hash ^ this.zobrist[move.row][move.col][player];
        
        let score;
        if (i === 0) {
            score = -this.pvs(board, depth - 1, -beta, -alpha, 3 - player, newHash);
        } else {
            // 零窗口搜索
            score = -this.pvs(board, depth - 1, -alpha - 1, -alpha, 3 - player, newHash);
            if (score > alpha && score < beta) {
                // 重新搜索
                score = -this.pvs(board, depth - 1, -beta, -score, 3 - player, newHash);
            }
        }
        
        // 撤销走法
        board[move.row][move.col] = 0;
        
        if (score > bestScore) {
            bestScore = score;
        }
        
        if (score > alpha) {
            alpha = score;
            flag = 'exact';
            this.historyTable[move.row][move.col] += depth * depth;
        }
        
        if (alpha >= beta) {
            flag = 'lower';
            this.searchStats.cutoffs++;
            break;
        }
    }
    
    // 存入置换表
    this.transpositionTable.set(hashStr, {
      score: bestScore,
      depth: depth,
      flag: flag,
      age: this.tableAge,
      pieceCount: currentPieceCount
    });
    
    return bestScore;
}
```

```javascript
getBestMove(board, player, timeLimit = null) {
    // 输入验证
    if (!board || !Array.isArray(board) || board.length === 0) {

    if (pieceCount >= this.size * this.size) {
        console.log('AI: 棋盘已满，无法落子');

    }
    
    // 初始化搜索统计
    this.searchStats = {
        nodes: 0,
        cutoffs: 0,
        cacheHits: 0,
        cacheMisses: 0,
        evaluations: 0,
        pieceCount: pieceCount,
        complexity: complexity
    };
    
    // 动态调整时间限制
    let adjustedTimeLimit = timeLimit || this.timeLimitConfig || 1000;
    
    if (complexity < 0.1) {
        // 开局阶段
    } else if (complexity < 0.25) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.3);
    } else if (complexity < 0.6) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 0.8);
    } else if (complexity < 0.8) {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.3);
    } else {
        adjustedTimeLimit = Math.floor(adjustedTimeLimit * 1.5);
    }
    
    this.timeLimit = adjustedTimeLimit;
    this.startTime = Date.now();
    this.nodeCount = 0;
    this.tableAge++;
    
    // 清理过旧的置换表项
    if (this.transpositionTable.size > this.maxTableSize) {
        this.cleanTranspositionTable();
    }
    
    // 重置历史启发表
    this.historyTable = this.createArray2D(0);
    
    // 检查开局库
    const bookMove = this.checkOpeningBook(board, player);
    if (bookMove) {
        console.log('AI: 使用开局库', bookMove);
        return bookMove;
    }
    
    // 终局优化
    if (complexity > 0.7) {
        this.depthConfig.max += 1;
        console.log('AI: 终局阶段，增加搜索深度');
    }

    // 终局多重威胁检测
    if (complexity > 0.7) {
        const multipleThreats = this.detectMultipleThreats(board, player);
        if (multipleThreats.length > 0) {
            const winningThreat = multipleThreats.find(t =>
                t.attackScore >= this.patternScores.LIVE_FOUR || 
                t.attackScore >= this.patternScores.RUSH_FOUR
            );
            if (winningThreat) {
                console.log('AI: 终局找到必胜走法', winningThreat);
                return winningThreat;
            }
            const defenseThreat = multipleThreats.find(t => 
                t.defenseScore >= this.patternScores.LIVE_FOUR
            );
            if (defenseThreat) {
                console.log('AI: 终局必须防守', defenseThreat);
                return defenseThreat;
            }
        }
    }

    let bestMove = null;
    let bestScore = -Infinity;
    
    // 迭代加深搜索
    for (let depth = 2; depth <= this.depthConfig.max; depth += 2) {
        const elapsedTime = Date.now() - this.startTime;
        const remainingTime = this.timeLimit - elapsedTime;
        
        if (remainingTime < this.timeLimit * 0.2) {
            console.log('AI: 时间不足，停止迭代加深');
            break;
        }
        
        if (elapsedTime > this.timeLimit * 0.8) {
            console.log('AI: 时间已用80%，停止迭代加深');
            break;
        }
        
        const result = this.searchRoot(board, player, depth);
        
        if (result && result.move) {
            bestMove = result.move;
            bestScore = result.score;
            
            if (bestScore >= this.patternScores.FIVE) {
                break;
            }
        }
        
        if (depth >= this.depthConfig.max) {
            break;
        }
    }
    
    // 如果没有找到走法，下在中心
    if (!bestMove) {
        const center = Math.floor(this.size / 2);
        if (board[center][center] === 0) {
            bestMove = { row: center, col: center };
        } else {
            for (let i = 0; i < this.size; i++) {
                for (let j = 0; j < this.size; j++) {
                    if (board[i][j] === 0) {
                        bestMove = { row: i, col: j };
                        break;
                    }
                }
                if (bestMove) break;
            }
        }
    }
    
    const searchTime = Date.now() - this.startTime;
    console.log('AI: 搜索完成', {
        move: bestMove,
        time: `${searchTime}ms`,
        stats: {
            nodes: this.searchStats.nodes,
            cutoffs: this.searchStats.cutoffs,
            cacheHits: this.searchStats.cacheHits,
            cacheMisses: this.searchStats.cacheMisses,
            evaluations: this.searchStats.evaluations
        }
    });
    
    return bestMove;
}
```

## 核心算法总结

### AI 算法核心特点

1. **Alpha-Beta 剪枝 + PVS（主变搜索）**
   - 使用零窗口搜索提高效率
   - 迭代加深搜索

2. **Zobrist 置换表**
   - 使用 LRU 缓存存储搜索结果
   - 避免重复计算相同局面
   - 最大容量 50MB

3. **历史启发**
   - 记录走法历史，优先搜索好的走法
   - 提高剪枝效率

4. **候选点剪枝**
   - 只搜索已有棋子周围 2-3 格内的空位
